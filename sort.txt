void sort(void) {
    int i;
    int j;
    int temp;
    i = 0;
    while (i < 9) {
        j = i + 1;
        while (j < 10) {
            if (arr[j] < arr[i]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
            j = j + 1;
        }
        i = i + 1;
    }
}
